1. ifconfig; The IP address is listed after inet, under eth0
2. nslookup [HOSTNAME]
3. host [IP ADDRESS]
4. scp -r [LOCAL FILEPATH] [USER@REMOTE SERVER:REMOTE FILEPATH]
5. netstat -ln | grep:[PORT]
6. netstat -ln
7. wget -r [WEBSITE URL]
8. curl -I [WEBSITE URL]
9. curl -Iv [WEBSITE URL]
